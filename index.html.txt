    <script>
        const hardwareData = [
            { name: "Fare", cat: "Giriş Birimi", color: "#3b82f6" },
            { name: "Klavye", cat: "Giriş Birimi", color: "#3b82f6" },
            { name: "Tarayıcı", cat: "Giriş Birimi", color: "#3b82f6" },
            { name: "Mikrofon", cat: "Giriş Birimi", color: "#3b82f6" },
            { name: "Webcam", cat: "Giriş Birimi", color: "#3b82f6" },
            { name: "Ekran", cat: "Çıkış Birimi", color: "#ef4444" },
            { name: "Yazıcı", cat: "Çıkış Birimi", color: "#ef4444" },
            { name: "Hoparlör", cat: "Çıkış Birimi", color: "#ef4444" },
            { name: "Kulaklık", cat: "Çıkış Birimi", color: "#ef4444" },
            { name: "SSD", cat: "Depolama", color: "#10b981" },
            { name: "USB Bellek", cat: "Depolama", color: "#10b981" },
            { name: "Sabit Disk", cat: "Depolama", color: "#10b981" },
            { name: "İşlemci", cat: "İç Donanım", color: "#f59e0b" },
            { name: "RAM", cat: "İç Donanım", color: "#f59e0b" },
            { name: "Anakart", cat: "İç Donanım", color: "#f59e0b" },
            { name: "Ekran Kartı", cat: "İç Donanım", color: "#f59e0b" }
        ];

        const categories = ["Giriş Birimi", "Çıkış Birimi", "Depolama", "İç Donanım"];
        
        let score = 0;
        let timeLeft = 60;
        let gameActive = false;
        let currentTargetCat = "";
        let spawnInterval;
        let timerInterval;

        const gameArea = document.getElementById('game-area');
        const viewport = document.getElementById('game-viewport');
        const scoreEl = document.getElementById('score');
        const timerEl = document.getElementById('timer');
        const targetEl = document.getElementById('target-category');
        const overlay = document.getElementById('overlay');
        const startBtn = document.getElementById('start-btn');
        const overlayTitle = document.getElementById('overlay-title');
        const overlayDesc = document.getElementById('overlay-desc');

        function initGame() {
            score = 0;
            timeLeft = 60;
            gameActive = true;
            scoreEl.innerText = score;
            timerEl.innerText = timeLeft;
            overlay.classList.add('hidden');
            gameArea.innerHTML = "";
            
            changeTargetCategory();
            
            spawnInterval = setInterval(spawnBubble, 1100);
            timerInterval = setInterval(updateTimer, 1000);
        }

        function changeTargetCategory() {
            currentTargetCat = categories[Math.floor(Math.random() * categories.length)];
            targetEl.innerText = currentTargetCat.toUpperCase();
            
            // Subtle transition effect
            targetEl.parentElement.animate([
                { transform: 'scale(1)', opacity: 0.5 },
                { transform: 'scale(1.05)', opacity: 1 },
                { transform: 'scale(1)', opacity: 1 }
            ], { duration: 300 });
        }

        function updateTimer() {
            timeLeft--;
            timerEl.innerText = timeLeft;
            if (timeLeft <= 0) endGame();
            if (timeLeft % 12 === 0 && timeLeft !== 60) changeTargetCategory();
        }

        function spawnBubble() {
            if (!gameActive) return;

            const data = hardwareData[Math.floor(Math.random() * hardwareData.length)];
            const bubble = document.createElement('div');
            bubble.className = 'bubble';
            bubble.innerText = data.name;
            
            const size = 90 + Math.random() * 30;
            bubble.style.width = `${size}px`;
            bubble.style.height = `${size}px`;
            bubble.style.backgroundColor = data.color + "cc";
            
            const startX = Math.random() * (viewport.offsetWidth - size);
            bubble.style.left = `${startX}px`;
            bubble.style.top = `-${size}px`;
            
            gameArea.appendChild(bubble);

            const duration = 5000 + Math.random() * 2000;
            const animation = bubble.animate([
                { top: `-${size}px` },
                { top: `${viewport.offsetHeight}px` }
            ], {
                duration: duration,
                easing: 'linear'
            });

            animation.onfinish = () => bubble.remove();

            bubble.addEventListener('click', (e) => {
                if (!gameActive) return;
                handleCapture(data, bubble, e);
            });
        }

        function handleCapture(item, element, event) {
            const rect = viewport.getBoundingClientRect();
            const x = event.clientX - rect.left;
            const y = event.clientY - rect.top;
            
            createParticles(x, y, item.color);
            
            if (item.cat === currentTargetCat) {
                score += 10;
                scoreEl.innerText = score;
                element.style.transform = 'scale(0) rotate(180deg)';
            } else {
                score = Math.max(0, score - 5);
                scoreEl.innerText = score;
                element.style.filter = 'grayscale(1) contrast(0.5)';
                element.style.opacity = '0';
            }

            setTimeout(() => element.remove(), 200);
        }

        function createParticles(x, y, color) {
            for (let i = 0; i < 10; i++) {
                const p = document.createElement('div');
                p.className = 'particle';
                p.style.width = '6px';
                p.style.height = '6px';
                p.style.backgroundColor = color;
                p.style.left = x + 'px';
                p.style.top = y + 'px';
                gameArea.appendChild(p);

                const angle = Math.random() * Math.PI * 2;
                const dist = 60 + Math.random() * 40;
                
                p.animate([
                    { transform: 'translate(0,0) scale(1)', opacity: 1 },
                    { transform: `translate(${Math.cos(angle) * dist}px, ${Math.sin(angle) * dist}px) scale(0)`, opacity: 0 }
                ], {
                    duration: 600,
                    easing: 'ease-out'
                }).onfinish = () => p.remove();
            }
        }

        function endGame() {
            gameActive = false;
            clearInterval(spawnInterval);
            clearInterval(timerInterval);
            
            document.querySelectorAll('.bubble').forEach(b => b.remove());
            
            overlayTitle.innerText = "OYUN BİTTİ!";
            overlayDesc.innerHTML = `Mükemmel bir iş çıkardın!<br><span class="text-white text-5xl block my-4 font-black">${score} Puan</span>`;
            startBtn.innerText = "YENİDEN DENE";
            overlay.classList.remove('hidden');
        }

        startBtn.addEventListener('click', initGame);
    </script>
